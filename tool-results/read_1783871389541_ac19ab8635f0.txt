   787→    def _pick_tp_levels(direction: str, entry_price: float | None, candidates: list[float]) -> tuple[float | None, float | None, float | None]:
   788→        if entry_price is None or not candidates:
   789→            return None, None, None
   790→
   791→        direction = (direction or "").lower().strip()
   792→        if direction not in ("long", "short"):
   793→            return None, None, None
   794→
   795→        uniq = sorted(set(round(x, 6) for x in candidates if x is not None and abs(x - entry_price) > 1e-6))
   796→        if not uniq:
   797→            return None, None, None
   798→
   799→        if direction == "long":
   800→            ordered = [x for x in uniq if x > entry_price]
   801→            if not ordered:
   802→                return None, None, None
   803→            tp1 = ordered[0]
   804→            tp2 = ordered[1] if len(ordered) > 1 else None
   805→            tp3 = ordered[2] if len(ordered) > 2 else None
   806→            if tp2 is None:
   807→                tp2 = tp1
   808→            if tp3 is None:
   809→                tp3 = tp2 if tp2 != tp1 else None
   810→            return tp1, tp2, tp3
   811→
   812→        ordered = [x for x in reversed(uniq) if x < entry_price]
   813→        if not ordered:
   814→            return None, None, None
   815→        tp1 = ordered[0]
   816→        tp2 = ordered[1] if len(ordered) > 1 else None
   817→        tp3 = ordered[2] if len(ordered) > 2 else None
   818→        if tp2 is None:
   819→            tp2 = tp1
   820→        if tp3 is None:
   821→            tp3 = tp2 if tp2 != tp1 else None
   822→        return tp1, tp2, tp3

   824→    def _calc_rr(entry_price: float | None, sl: float | None, tp1: float | None) -> float | None:
   825→        if entry_price is None or sl is None or tp1 is None:
   826→            return None
   827→        risk = abs(entry_price - sl)
   828→        reward = abs(tp1 - entry_price)
   829→        if risk <= 0 or reward <= 0:
   830→            return None
   831→        return round(reward / risk, 2)

  1748→    if not levels:

  1751→    merged = [levels[0]]
  1752→    for lvl in levels[1:]:

  1943→def _format_zigzag_context_compact(ctx: dict) -> str:
  1944→    """
  1945→    Компактный текстовый сериализатор ZigZag контекста для промпта LLM.
  1946→    ~700 chars вместо 3584 chars сырого JSON — меньше токенов, больше сигнала.
  1947→    """
  1948→    if not ctx or not isinstance(ctx, dict) or ctx.get("error"):
  1949→        return ctx.get("message", "ZigZag недоступен.") if isinstance(ctx, dict) else "{}"
  1950→
  1951→    lines = []
  1952→
  1953→    # Stack summary
  1954→    stack = ctx.get("stack") or {}
  1955→    if stack:
  1956→        lines.append(
  1957→            f"Stack: bias={stack.get('stack_bias','?')} align={stack.get('alignment','?')} "
  1958→            f"dom={stack.get('dominant_tf','?')} | {' '.join(f'{k}:{v}' for k,v in (stack.get('directions') or {}).items())}"
  1959→        )
  1960→
  1961→    # Per-TF zones
  1962→    tfs = ctx.get("timeframes") or {}
  1963→    for tf, data in tfs.items():
  1964→        if not isinstance(data, dict):
  1965→            continue
  1966→        lines.append(
  1967→            f"{tf}: [{data.get('lower','?')}–{data.get('upper','?')}] "
  1968→            f"mode={data.get('market_mode','?')} swing={data.get('swing_direction','?')} "
  1969→            f"pivots={data.get('pivot_count','?')} pos={data.get('price_position','?')}"
  1970→        )
  1971→
  1972→    # Confluence levels
  1973→    confluence = ctx.get("confluence_levels") or []
  1974→    if confluence:
  1975→        cf_str = ", ".join(
  1976→            f"{c.get('level','?')}({'+'.join(c.get('timeframes',[]))},{c.get('priority','?')})"
  1977→            for c in confluence[:6]
  1978→        )
  1979→        lines.append(f"Confluence: {cf_str}")
  1980→
  1981→    return "\n".join(lines) if lines else "ZigZag недоступен."
