    90→//--- state
    91→var array<Pool>       pools         = array.new<Pool>()
    92→var array<HistMagnet> history       = array.new<HistMagnet>()
    93→var array<label>      histLabels    = array.new<label>()

   100→var float prevTopPrice   = na
   101→var bool  prevTopIsHigh  = false
   102→
   103→atrV   = ta.atr(14)
   104→momRaw = ta.ema(close, 8) - ta.ema(close, 21)
   105→volMa  = ta.sma(volume, 50)
   106→
   107→//--- auto-scale by tf
   108→tfSec = timeframe.in_seconds(timeframe.period)
   109→
   110→autoLogD = 0.30
   111→if tfSec >= 604800
   112→    autoLogD := 0.95
   113→else if tfSec >= 86400
   114→    autoLogD := 0.65
   115→else if tfSec >= 14400
   116→    autoLogD := 0.45
   117→else if tfSec >= 3600
   118→    autoLogD := 0.30
   119→else if tfSec >= 900
   120→    autoLogD := 0.18
   121→else
   122→    autoLogD := 0.10
   123→
   124→logDistMax = autoScale ? autoLogD : manualLogD
   125→
   126→autoEqPct = 0.15
   127→if tfSec >= 604800
   128→    autoEqPct := 10
   129→else if tfSec >= 86400
   130→    autoEqPct := 8
   131→else if tfSec >= 14400
   132→    autoEqPct := 5
   133→else if tfSec >= 3600
   134→    autoEqPct := 0.8
   135→else if tfSec >= 900
   136→    autoEqPct := 0.5
   137→else
   138→    autoEqPct := 0.15
   139→
   140→eqTolPct = (autoEqTol ? autoEqPct : manualEqTol) / 100.0
   141→
   142→//--- helpers
   143→logDist(float a, float b) =>
   144→    (a > 0 and b > 0) ? math.abs(math.log(a / b)) : 1e10
   145→
   146→findClosestPool(float newPrice, bool isHigh) =>
   147→    idx     = -1
   148→    minDist = 1e10
   149→    sz      = array.size(pools)
   150→    if sz > 0
   151→        for i = 0 to sz - 1
   152→            p = array.get(pools, i)
   153→            if p.isHigh == isHigh
   154→                diff = math.abs(p.price - newPrice) / newPrice
   155→                if diff <= eqTolPct and diff < minDist
   156→                    minDist := diff
   157→                    idx     := i
   158→    idx
   159→
   160→clearPoolDraws(Pool p) =>
   161→    if not na(p.bx)
   162→        box.delete(p.bx)
   163→    if not na(p.ln)
   164→        line.delete(p.ln)
   165→    if not na(p.lbl)
   166→        label.delete(p.lbl)
   167→
   168→strToSize(string s) =>
   169→    result = size.small
   170→    if s == "Tiny"
   171→        result := size.tiny
   172→    else if s == "Small"
   173→        result := size.small
   174→    else if s == "Normal"
   175→        result := size.normal
   176→    else if s == "Large"
   177→        result := size.large
   178→    else if s == "Huge"
   179→        result := size.huge
   180→    result
   181→
   182→strToPos(string s) =>
   183→    result = position.top_right
   184→    if s == "Top Right"
   185→        result := position.top_right
   186→    else if s == "Top Left"
   187→        result := position.top_left
   188→    else if s == "Bottom Right"
   189→        result := position.bottom_right
   190→    else if s == "Bottom Left"
   191→        result := position.bottom_left
   192→    else if s == "Middle Right"
   193→        result := position.middle_right
   194→    else if s == "Middle Left"
   195→        result := position.middle_left
   196→    result
   197→
   198→dashSize   = strToSize(dashSizeStr)
   199→lblSize    = strToSize(lblSizeStr)
   200→topLblSize = strToSize(topLblSizeStr)
   201→magSize    = strToSize(magnetSize)
   202→histSize   = strToSize(histMarkerSize)
   203→dashPos    = strToPos(dashPosStr)
   204→
   205→//--- pivot detect + merge
   206→ph = ta.pivothigh(high, pivotLen, pivotLen)
   207→pl = ta.pivotlow(low,  pivotLen, pivotLen)
   208→
   209→if not na(ph)
   210→    mIdx = findClosestPool(ph, true)
   211→    if mIdx >= 0
   212→        existing = array.get(pools, mIdx)
   213→        existing.isEqual := true
   214→        existing.price   := math.max(existing.price, ph)
   215→        existing.volAt   := existing.volAt + volume[pivotLen]
   216→        array.set(pools, mIdx, existing)
   217→    else
   218→        array.push(pools, Pool.new(ph, bar_index - pivotLen, true, false, volume[pivotLen], 0.0, 0.0, na, na, na))
   219→
   220→if not na(pl)
   221→    mIdx = findClosestPool(pl, false)
   222→    if mIdx >= 0
   223→        existing = array.get(pools, mIdx)
   224→        existing.isEqual := true
   225→        existing.price   := math.min(existing.price, pl)
   226→        existing.volAt   := existing.volAt + volume[pivotLen]
   227→        array.set(pools, mIdx, existing)
   228→    else
   229→        array.push(pools, Pool.new(pl, bar_index - pivotLen, false, false, volume[pivotLen], 0.0, 0.0, na, na, na))
   230→
   231→//--- prune
   232→prevTopWasSweptThisBar = false
   233→
   234→ii = array.size(pools) - 1
   235→while ii >= 0
   236→    p      = array.get(pools, ii)
   237→    age    = bar_index - p.barIdx
   238→    swept  = (p.isHigh and high > p.price) or (not p.isHigh and low < p.price)
   239→    tooOld = age > maxAge
   240→    lDist  = logDist(p.price, close)
   241→    tooFar = lDist > logDistMax
   242→
   243→    isPrevTop = not na(prevTopPrice) and math.abs(p.price - prevTopPrice) < syminfo.mintick * 5 and p.isHigh == prevTopIsHigh
   244→
   245→    if swept
   246→        if isPrevTop
   247→            prevTopWasSweptThisBar := true
   248→        clearPoolDraws(p)
   249→        array.remove(pools, ii)
   250→    else if tooOld
   251→        clearPoolDraws(p)
   252→        array.remove(pools, ii)
   253→    else if tooFar
   254→        clearPoolDraws(p)
   255→        array.remove(pools, ii)
   256→    ii -= 1
   257→
   258→while array.size(pools) > maxPools
   259→    pShift = array.shift(pools)
   260→    clearPoolDraws(pShift)
   261→
   262→//--- scoring
   263→calcScore(Pool p) =>
   264→    age   = bar_index - p.barIdx
   265→    lDist = logDist(p.price, close)
   266→
   267→    volFactor = (na(p.volAt) or volMa <= 0) ? 1.0 : math.min(p.volAt / volMa, 3.0)
   268→    eqMult    = p.isEqual ? eqBonus : 1.0
   269→    strength  = volFactor * eqMult
   270→
   271→    prox = 0.0
   272→    if lDist < 0.002
   273→        prox := 0.3
   274→    else if lDist < 0.05
   275→        prox := math.max(0.5, 1.0 - math.abs(lDist - 0.02) * 10.0)
   276→    else
   277→        prox := math.max(0.05, 1.0 - (lDist - 0.05) * 3.0)
   278→
   279→    ageScore = 0.0
   280→    if age < minAge
   281→        ageScore := 0.5
   282→    else if age < 50
   283→        ageScore := 1.0
   284→    else
   285→        ageScore := math.max(0.1, 1.0 - (age - 50) / 450.0)
   286→
   287→    dirToPool = p.price > close ? 1.0 : -1.0
   288→    momNorm   = close > 0 ? momRaw / close : 0.0
   289→    momAlign  = dirToPool * momNorm * 100.0
   290→    momScore  = 1.0 + math.max(-0.5, math.min(0.5, momAlign))
   291→
   292→    rawScore = strength * wStrength + prox * wProximity + ageScore * wAge + momScore * wMomentum
   293→    math.max(rawScore, 0.01)
   294→
   295→totalScore = 0.0
   296→scoreUp    = 0.0
   297→scoreDn    = 0.0
   298→topIdx     = -1
   299→topScore   = -1.0
   300→
   301→if array.size(pools) > 0
   302→    for i = 0 to array.size(pools) - 1
   303→        p = array.get(pools, i)
   304→        p.score := calcScore(p)
   305→        array.set(pools, i, p)
   306→        totalScore += p.score
   307→        if p.score > topScore
   308→            topScore := p.score
   309→            topIdx   := i
   310→
   311→    if totalScore > 0
   312→        for i = 0 to array.size(pools) - 1
   313→            p = array.get(pools, i)
   314→            p.probPct := (p.score / totalScore) * 100.0
   315→            array.set(pools, i, p)
   316→            if p.isHigh
   317→                scoreUp += p.score
   318→            else
   319→                scoreDn += p.score
   320→
   321→topPrice   = (topIdx >= 0 and array.size(pools) > 0) ? array.get(pools, topIdx).price   : na
   322→topProb    = (topIdx >= 0 and array.size(pools) > 0) ? array.get(pools, topIdx).probPct : na
   323→topIsHigh  = (topIdx >= 0 and array.size(pools) > 0) ? array.get(pools, topIdx).isHigh  : false
   324→
   325→//--- magnet pull
   326→biasMag = 0.0
   327→if not na(topPrice) and totalScore > 0
   328→    sameSidePct = topIsHigh ? scoreUp / totalScore * 100.0 : scoreDn / totalScore * 100.0
   329→    biasMag := sameSidePct
   330→biasDir = topIsHigh ? 1 : -1
   331→bias = biasMag * biasDir
   332→
   333→distToTopPct = na(topPrice) ? na : (topPrice - close) / close * 100.0
   334→
   335→//--- top 3 calculation
   336→type RankedPool
   337→    float price
   338→    float prob
   339→    bool  isHigh
   340→    bool  isEqual
   341→
   342→var array<RankedPool> top3 = array.new<RankedPool>()
   343→array.clear(top3)
   344→
   345→if array.size(pools) > 0
   346→    tmpPrices = array.new<float>()
   347→    tmpProbs  = array.new<float>()
   348→    tmpHighs  = array.new<bool>()
   349→    tmpEquals = array.new<bool>()
   350→    for i = 0 to array.size(pools) - 1
   351→        p = array.get(pools, i)
   352→        array.push(tmpPrices, p.price)
   353→        array.push(tmpProbs,  p.probPct)
   354→        array.push(tmpHighs,  p.isHigh)
   355→        array.push(tmpEquals, p.isEqual)
   356→
   357→    sz = array.size(tmpProbs)
   358→    n = math.min(3, sz)
   359→    if n > 0
   360→        for k = 0 to n - 1
   361→            bestIdx = k
   362→            if k + 1 <= sz - 1
   363→                for j = k + 1 to sz - 1
   364→                    if array.get(tmpProbs, j) > array.get(tmpProbs, bestIdx)
   365→                        bestIdx := j
   366→            if bestIdx != k
   367→                tp = array.get(tmpPrices, k)
   368→                array.set(tmpPrices, k, array.get(tmpPrices, bestIdx))
   369→                array.set(tmpPrices, bestIdx, tp)
   370→                tpr = array.get(tmpProbs, k)
   371→                array.set(tmpProbs, k, array.get(tmpProbs, bestIdx))
   372→                array.set(tmpProbs, bestIdx, tpr)
   373→                th = array.get(tmpHighs, k)
   374→                array.set(tmpHighs, k, array.get(tmpHighs, bestIdx))
   375→                array.set(tmpHighs, bestIdx, th)
   376→                te = array.get(tmpEquals, k)
   377→                array.set(tmpEquals, k, array.get(tmpEquals, bestIdx))
   378→                array.set(tmpEquals, bestIdx, te)
   379→            array.push(top3, RankedPool.new(array.get(tmpPrices, k), array.get(tmpProbs, k), array.get(tmpHighs, k), array.get(tmpEquals, k)))
   380→
   381→//--- historical track
   382→topActuallyChanged = not na(prevTopPrice) and not na(topPrice) and (math.abs(prevTopPrice - topPrice) > syminfo.mintick * 5 or prevTopIsHigh != topIsHigh)
   383→topDisappeared     = not na(prevTopPrice) and na(topPrice)
   384→
   385→if topActuallyChanged or topDisappeared
   386→    initialStatus = prevTopWasSweptThisBar ? 1 : 0
   387→    resBar        = prevTopWasSweptThisBar ? bar_index : na
   388→    array.push(history, HistMagnet.new(prevTopPrice, bar_index, prevTopIsHigh, initialStatus, resBar))
   389→    if prevTopWasSweptThisBar
   390→        touchCount += 1
   391→
   392→if array.size(history) > 0
   393→    for i = 0 to array.size(history) - 1
   394→        h = array.get(history, i)
   395→        if h.status == 0
   396→            barsElapsed = bar_index - h.recordedBar
   397→            touched = (h.isHigh and high >= h.price) or (not h.isHigh and low <= h.price)
   398→            if touched
   399→                h.status      := 1
   400→                h.resolvedBar := bar_index
   401→                touchCount    += 1
   402→                array.set(history, i, h)
   403→            else if barsElapsed >= trackingWindow
   404→                h.status      := 2
   405→                h.resolvedBar := bar_index
   406→                expiredCount  += 1
   407→                array.set(history, i, h)
   408→
   409→while array.size(history) > maxHistory
   410→    array.shift(history)
   411→
   412→prevTopPrice  := topPrice
   413→prevTopIsHigh := topIsHigh
   414→
   415→//--- visualization
   416→intensityTransp(float pct) =>
   417→    int(math.max(10, 85 - pct * 3.5))
   418→
   419→if barstate.islast and showZones and array.size(pools) > 0
   420→    atrH = atrV * 0.04
   421→    for i = 0 to array.size(pools) - 1
   422→        p = array.get(pools, i)
   423→        clearPoolDraws(p)
   424→        isTop = (i == topIdx) and showTop
