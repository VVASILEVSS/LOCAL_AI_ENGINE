   108→    try:
   109→        conn = sqlite3.connect(DB_PATH)
   110→        conn.row_factory = sqlite3.Row
   111→        c = conn.cursor()
   112→        c.execute(query, args)
   113→        rows = c.fetchall()
   114→        conn.close()
   115→        return [dict(r) for r in rows]
   116→    except Exception:
   117→        return []

   172→def clean_analysis_report(text: str) -> str:
   173→    """Исправляет логические противоречия в ТП/СЛ и форматирует вывод."""
   174→    if "НАПРАВЛЕНИЕ:" in text:
   175→        direction = "Long" if "Long" in text.split("НАПРАВЛЕНИЕ:")[1].split("|")[0] else "Short"
   176→        price_match = re.search(r"Текущая цена:\s*([\d.]+)", text)
   177→        if price_match:
   178→            current = float(price_match.group(1))
   179→            tp1_match = re.search(r"TP1:\s*([\d.]+)", text)
   180→            if tp1_match:
   181→                tp1 = float(tp1_match.group(1))
   182→                if direction == "Long" and tp1 <= current:
   183→                    text = text.replace(f"TP1: {tp1}", "TP1: Уже отработан")
   184→                elif direction == "Short" and tp1 >= current:
   185→                    text = text.replace(f"TP1: {tp1}", "TP1: Уже отработан")
   186→    return text

   313→# ── /scan (полная копия из handlers.py, но через cloud LLM) ────────────────
   314→
   315→def _fill_missing_tf_zones(
   316→    result: dict, prev_tf_zones: dict, timeframes: list[str],
   317→    zigzag_timeframes: dict | None = None,
   318→    vp_timeframes: dict | None = None,
   319→) -> None:
   320→    """
   321→    Если LLM не вернул зону для какого-то ТФ — берёт из fallback источников.
   322→
   323→    Приоритет fallback:
   324→    0. Volume Profile POC (настоящие зоны консолидации по объёму)
   325→    1. ZigZag benchmark
   326→    2. prev_tf_zones (для ТФ которых ZigZag/VP не загрузили)
   327→
   328→    Если нет нигде — зона не вставляется (N/A).
   329→    Если нет нигде — зона не вставляется (N/A).
   330→
   331→    НЕ используем all_metrics[tf]["zone"] = get_structural_extremums()
   332→    (сырые max/min за 120 свечей — не зоны консолидации).
   333→
   334→    Матрёшка + D1 cap применяются ВНЕ функции — через re-run enforce_risk_rules()
   335→    в caller (_do_full_scan) после fallback.
   336→    """
   337→    tf_key_map = {
   338→        "1d": "1D", "4h": "4H", "1h": "1H", "15m": "15M", "5m": "5M",
   339→        "1D": "1D", "4H": "4H", "1H": "1H", "15M": "15M", "5M": "5M",
   340→    }
   341→    # Канонический порядок от старшего к младшему
   342→    canonical_order = ["1D", "4H", "1H", "15M", "5M"]
   343→
   344→    if not isinstance(result, dict):
   345→        return
   346→    tf_zones = result.get("tf_zones")
   347→    if not isinstance(tf_zones, dict):
   348→        tf_zones = {}
   349→        result["tf_zones"] = tf_zones
   350→
   351→    if not isinstance(zigzag_timeframes, dict):
   352→        zigzag_timeframes = {}
   353→    if not isinstance(vp_timeframes, dict):
   354→        vp_timeframes = {}
   355→
   356→    filled = False
   357→    price = result.get("price") or result.get("current_price")
   358→    price_safe = price if isinstance(price, (int, float)) and price > 0 else None
   359→    stick_tol = 0.005  # 0.5% — если parent и child зоны совпадают в пределах tolerance
   360→
   361→    for tf in timeframes:
   362→        norm_key = tf_key_map.get(tf, tf.upper().replace("MIN", "M"))
   363→        if norm_key in tf_zones and isinstance(tf_zones[norm_key], dict):
   364→            # Проверка «прилипания»: если эта зона = parent зоне (D1=H4=H1)
   365→            # → LLM скопировал, заменяем на fallback (VP/ZigZag)
   366→            if price_safe and norm_key != "1D":
   367→                parent_idx = canonical_order.index(norm_key) - 1 if norm_key in canonical_order else -1
   368→                if parent_idx >= 0:
   369→                    parent_tf = canonical_order[parent_idx]
   370→                    parent_z = tf_zones.get(parent_tf)
   371→                    child_z = tf_zones[norm_key]
   372→                    if (isinstance(parent_z, dict) and isinstance(child_z, dict)
   373→                            and parent_z.get("lower") is not None and child_z.get("lower") is not None
   374→                            and parent_z.get("upper") is not None and child_z.get("upper") is not None):
   375→                        ldiff = abs(parent_z["lower"] - child_z["lower"]) / price_safe
   376→                        udiff = abs(parent_z["upper"] - child_z["upper"]) / price_safe
   377→                        if ldiff < stick_tol and udiff < stick_tol:
   378→                            logging.info(
   379→                                "DASHBOARD: %s zone sticks to %s (diff l=%.3f%% u=%.3f%%), replacing with fallback for %s",
   380→                                norm_key, parent_tf, ldiff * 100, udiff * 100, result.get("symbol", "?"),
   381→                            )
   382→                            # Удаляем прилипшую зону — fallback ниже подставит реальную
   383→                            del tf_zones[norm_key]
   384→            # После проверки — если зона всё ещё на месте, пропускаем
   385→            if norm_key in tf_zones and isinstance(tf_zones[norm_key], dict):
   386→                # Доп. проверка: min-span — если зона слишком узкая (микроканал), тоже удаляем
   387→                if price_safe:
   388→                    z = tf_zones[norm_key]
   389→                    z_upper = z.get("upper")
   390→                    z_lower = z.get("lower")
   391→                    if z_upper is not None and z_lower is not None:
   392→                        span_pct = abs(z_upper - z_lower) / price_safe
   393→                        # Минимальный span по ТФ: ниже = микроканал, не структурная зона
   394→                        # Пороги основаны на реальных структурных ranges (20 свечей Binance)
   395→                        # LLM часто ставит зону = последние 5 свечей в сжатии (0.3-0.8%)
   396→                        # Реальный 15m structural range ≈ 1-3% (D1≈5-10%, H4≈3-5%, H1≈1.5-3%)
   397→                        min_span = {"1D": 0.025, "4H": 0.020, "1H": 0.012, "15M": 0.008, "5M": 0.004}
   398→                        min_pct = min_span.get(norm_key, 0.002)
   399→                        if span_pct < min_pct:
   400→                            logging.info(
   401→                                "DASHBOARD: %s zone too narrow: %.4f%% < min %.4f%%, replacing with fallback for %s",
   402→                                norm_key, span_pct * 100, min_pct * 100, result.get("symbol", "?"),
   403→                            )
   404→                            del tf_zones[norm_key]
   405→                if norm_key in tf_zones and isinstance(tf_zones[norm_key], dict):
   406→                    continue
   407→
   408→        zone = None
   409→        source = None
   410→
   411→        # Поиск ключа: tf (raw), norm_key (canonical), tf_lower (VP/ZigZag return lowercase)
   412→        tf_lower = tf.lower()
   413→        lookup_keys = [tf, norm_key, tf_lower]
   414→
   415→        def _find_zone(d):
   416→            """Ищем зону по любому варианту ключа (case-insensitive)."""
   417→            if not isinstance(d, dict):
   418→                return None
   419→            for k in lookup_keys:
   420→                v = d.get(k)
   421→                if isinstance(v, dict) and (v.get("upper") is not None or v.get("lower") is not None):
   422→                    return v
   423→            return None
   424→
   425→        # Fallback-0: Volume Profile POC (настоящие зоны консолидации по объёму)
   426→        vp = _find_zone(vp_timeframes)
   427→        if vp is not None:
   428→            zone = {"upper": vp.get("upper"), "lower": vp.get("lower")}
   429→            source = "volume_profile"
   430→
   431→        # Fallback-1: ZigZag benchmark
   432→        if zone is None:
   433→            zz = _find_zone(zigzag_timeframes)
   434→            if zz is not None:
   435→                zone = {"upper": zz.get("upper"), "lower": zz.get("lower")}
   436→                source = "zigzag"
   437→
   438→        # Fallback-2: prev_tf_zones (для ТФ которых ZigZag/VP не загрузили, напр. 1D fetch failed)
   439→        if zone is None:
   440→            prev_zone = _find_zone(prev_tf_zones)
   441→            if prev_zone is not None:
   442→                zone = {"upper": prev_zone.get("upper"), "lower": prev_zone.get("lower")}
   443→                source = "prev_analysis"
   444→
   445→        if zone is not None:
   446→            tf_zones[norm_key] = zone
   447→            filled = True
   448→            logging.info(
   449→                "DASHBOARD: filled missing %s zone from %s for %s",
   450→                norm_key, source, result.get("symbol", "?"),
   451→            )
   452→
   453→    # Пересобрать tf_zones в каноническом порядке (D1→H4→H1→M15→M5).
   454→    # Python 3.7+ dict сохраняет порядок вставки — без этого D1 будет в конце.
   455→    if filled:
   456→        ordered = {}
   457→        for canon_tf in canonical_order:
   458→            z = tf_zones.get(canon_tf)
   459→            if isinstance(z, dict):
   460→                ordered[canon_tf] = z
   461→        # Добавить любые другие ключи, которых нет в canonical_order
   462→        for k, v in tf_zones.items():
   463→            if k not in ordered:
   464→                ordered[k] = v
   465→        result["tf_zones"] = ordered
   466→
   467→    # Пересчитать tf_span_map
   468→    if filled:
   469→        span_map = {}
   470→        for tf in timeframes:
   471→            norm_key = tf_key_map.get(tf, tf.upper().replace("MIN", "M"))
   472→            z = ordered.get(norm_key) if filled else tf_zones.get(norm_key)
   473→            if isinstance(z, dict):
   474→                upper = z.get("upper")
   475→                lower = z.get("lower")
   476→                if upper is not None and lower is not None:
   477→                    span_map[norm_key] = abs(upper - lower)
   478→        result["tf_span_map"] = span_map
   479→

   481→async def _do_full_scan(symbol: str, timeframes: list[str], chat_id: int, bot: Bot) -> None:
   482→    """Полный анализ символ/ТФ через cloud LLM — графики последовательно."""
   483→    async with SCAN_LOCK:
   484→        await bot.send_message(chat_id, f"📡 Загружаю {_format_symbol(symbol)} по ТФ: {', '.join(timeframes)}...")
   485→        try:
   486→            chart_bytes_list: list[bytes] = []
   487→            all_metrics: dict[str, dict] = {}
   488→
   489→            # Динамический limit по ТФ: старшим нужно больше свечей,
   490→            # чтобы видеть предыдущую + текущую структуру.
   491→            # H4: 300 свечей = 50 дней — виден предыдущий HH/HL

   494→                "1d": 250, "4h": 300, "1h": 200,
   495→                "15m": 150, "5m": 120,
   496→            }
   497→
   498→            # Графики ПОСЛЕДОВАТЕЛЬНО (один за другим)
   499→            for tf in timeframes:
   500→                limit = TF_LIMITS.get(tf.lower(), 120)
   501→                chart_bytes, metrics = fetch_and_plot(symbol=symbol, timeframe=tf, limit=limit)
   502→                chart_bytes_list.append(chart_bytes)
   503→                all_metrics[tf] = metrics
   504→
   505→            m_htf = all_metrics[timeframes[0]]
   506→            m_ltf = all_metrics[timeframes[-1]]
   507→
   508→            fib = m_ltf.get("fib_context", {"50%": "N/A", "61.8%": "N/A", "38.2%": "N/A", "rule": ""})
   509→            tf_zones = {tf: all_metrics[tf]["zone"] for tf in timeframes}
   510→
   511→            live_price = m_ltf.get("current_price", m_ltf.get("last_closed_price", 0))
   512→
   513→            tf_context = (
   514→                f"[HTF] {m_htf.get('phase', 'N/A')} | Упор: {m_htf.get('resistance', 'N/A')} | Поддержка: {m_htf.get('support', 'N/A')} | "
   515→                f"[{timeframes[-1]}] {m_ltf.get('phase', 'N/A')} | Текущая цена: {live_price} | "
   516→                f"Объём: {m_ltf.get('vol_ratio', 1.0)}x ({m_ltf.get('vol_trend', 'N/A')}) | "
   517→                f"Фибо: 50%={fib['50%']} | 61.8%={fib['61.8%']} | 38.2%={fib['38.2%']}"
   518→            )
   519→
   520→            stats = get_backtest_stats()
   521→            metrics_str = (
   522→                f"Текущая цена: {live_price} | Последняя закрытая: {m_ltf.get('last_closed_price', 'N/A')} | "
   523→                f"ATR: {m_ltf.get('atr', 'N/A')} | RSI: {m_ltf.get('rsi', 'N/A')} | Сессия: {m_ltf.get('session', 'N/A')}"
   524→            )
   525→
   526→            # ZigZag compact context
   527→            try:
   528→                from core.zigzag.benchmark_zigzag import run_benchmark
   529→                zigzag_benchmark = run_benchmark(
   530→                    symbol=symbol, market_type="future",
   531→                    timeframes=timeframes, limit=200,
   532→                    mode="hybrid_atr", confirmation_mode="close",
   533→                    debug=False, output=None, output_mode="compact",
   534→                )
   535→                zigzag_context = {
   536→                    "symbol": zigzag_benchmark.get("symbol", symbol),
   537→                    "normalized_symbol": zigzag_benchmark.get("normalized_symbol", symbol),
   538→                    "stack": zigzag_benchmark.get("stack", {}),
   539→                    "timeframes": zigzag_benchmark.get("timeframes", {}),
   540→                    "confluence_levels": zigzag_benchmark.get("confluence_levels", []),
   541→                }
   542→            except Exception as e:
   543→                zigzag_context = {
   544→                    "error": True, "message": f"ZigZag: {type(e).__name__}",
   545→                    "symbol": symbol, "stack": {}, "timeframes": {}, "confluence_levels": [],
   546→                }
   547→
   548→            # Volume Profile POC — настоящие зоны консолидации по объёму
   549→            try:
   550→                from core.volume_profile import run_volume_profile
   551→                vp_result = run_volume_profile(
   552→                    symbol=symbol, timeframes=timeframes,
   553→                    limit=200, bins=50, value_area_pct=0.70,
   554→                    market_type="future",
   555→                )
   556→                vp_context = {
   557→                    "symbol": vp_result.get("symbol", symbol),
   558→                    "timeframes": vp_result.get("timeframes", {}),
   559→                }
   560→            except Exception as e:
   561→                vp_context = {
   562→                    "error": True, "message": f"VP: {type(e).__name__}",
   563→                    "symbol": symbol, "timeframes": {},
   564→                }
   565→
   566→            ltf_volume = all_metrics[timeframes[-1]].get("volume_context", {})
   567→            if not isinstance(ltf_volume, dict):
   568→                ltf_volume = {}
   569→
   570→            # Liquidity heatmap
   571→            try:
   572→                from core.liquidity_heatmap import build_liquidity_context_text, build_liquidity_heatmap
   573→                from core.data_provider import OhlcvDataProvider
   574→                provider = OhlcvDataProvider()
   575→                ltf_tf = timeframes[-1]
   576→                try:
   577→                    ltf_df = provider.read_current_csv(symbol, ltf_tf)
   578→                    hm = build_liquidity_heatmap(ltf_df, symbol=symbol, timeframe=ltf_tf)
   579→                    heatmap_text = build_liquidity_context_text(hm)
   580→                except FileNotFoundError:
   581→                    heatmap_text = "Liquidity heatmap: CSV недоступен."
   582→            except Exception:
   583→                heatmap_text = "Liquidity heatmap: ошибка."
   584→
   585→            prev_ctx = {
   586→                "metrics": metrics_str,
   587→                "tf_context": tf_context,
   588→                "backtest": f"Win Rate: {stats['win_rate']}%, MAE: {stats['mae_pct']}%",
   589→                "tf_zones": tf_zones,
   590→                "zigzag_context": zigzag_context,
   591→                "volume_context": ltf_volume,
   592→                "heatmap_context": heatmap_text,
   593→            }
   594→
   595→            # LLM через CLOUD
   596→            raw_result = await analyze_multi_images(
   597→                chart_bytes_list,
   598→                prev_analysis=prev_ctx,
   599→                llm_api_key=DASHBOARD_LLM_API_KEY,
   600→                llm_base_url=DASHBOARD_LLM_BASE_URL,
   601→                llm_model=DASHBOARD_MODEL_NAME,
   602→            )
   603→
   604→            parsed_result = raw_result
   605→            if isinstance(parsed_result, dict):
   606→                parsed_result = update_and_save_state(symbol, timeframes[-1], parsed_result)
   607→
   608→            # enforce_risk_rules уже вызван внутри analyze_multi_images —
   609→            # tf_zones, D1 cap, nesting, confluence — всё валидировано.
   610→            # Но LLM может не вернуть зону для какого-то ТФ (часто 1D) —
   611→            # заполняем отсутствующие из prev_analysis.tf_zones как fallback.
   612→            # ВАЖНО: после fallback перезапускаем enforce_risk_rules,
   613→            # чтобы матрёшка и D1 cap применились к обновлённым зонам.
   614→            if isinstance(parsed_result, dict):
   615→                # Гарантировать symbol в result для логирования fallback
   616→                if not parsed_result.get("symbol"):
   617→                    parsed_result["symbol"] = symbol
   618→                _fill_missing_tf_zones(
   619→                    parsed_result, tf_zones, timeframes,
   620→                    zigzag_timeframes=zigzag_context.get("timeframes", {}),
   621→                    vp_timeframes=vp_context.get("timeframes", {}),
   622→                )
   623→                # Re-run enforce_risk_rules: применит матрёшку + D1 cap
   624→                # к fallback-зонам, добавленным из prev_analysis.
   625→                parsed_result = enforce_risk_rules(parsed_result)
   626→
   627→            if isinstance(parsed_result, dict) and parsed_result.get("error"):
   628→                await bot.send_message(chat_id, f"⚠️ Ошибка анализа: {parsed_result.get('message')}")
   629→                return
   630→
   631→            if isinstance(parsed_result, dict):
   632→                final_text = format_json_for_tg(parsed_result)
   633→            else:
   634→                final_text = str(parsed_result)
   635→
   636→            USER_ANALYSIS_CACHE[chat_id] = final_text
   637→            await bot.send_message(chat_id, f"📊 Анализ {_format_symbol(symbol)}:\n\n{final_text}")
   638→
   639→        except Exception as e:
   640→            await bot.send_message(chat_id, f"⚠️ Ошибка: {e}")

   643→@dp.message(Command("scan"))
   644→async def cmd_scan(message: Message) -> None:
   645→    if message.from_user and message.from_user.id != ADMIN_CHAT_ID: return
   646→    if not DASHBOARD_LLM_API_KEY:
   647→        await message.answer("❌ DASHBOARD_LLM_API_KEY не задан в .env")
   648→        return
   649→    if message.text is None:
   650→        return
   651→    # Support multi-symbol: /scan BTC | ETH | XAUT  OR  /scan BTC,ETH,XAUT  OR  /scan BTC ETH XAUT  OR  /scan BTC
   652→    raw_text = message.text.split(maxsplit=1)
   653→    raw_syms_str = raw_text[1] if len(raw_text) > 1 else "BTC"
   654→    # Normalize separators: '|' and ',' and whitespace (multiple spaces) → split
   655→    # /scan BTC | ETH | XAUT  →  ["BTC", "ETH", "XAUT"]
   656→    # /scan BTC,ETH,XAUT     →  ["BTC", "ETH", "XAUT"]
   657→    # /scan BTC ETH XAUT     →  ["BTC", "ETH", "XAUT"]
   658→    raw_syms_str_norm = raw_syms_str.replace("|", " ").replace(",", " ")
   659→    raw_syms = [s.strip().upper().replace("/", "").replace("USDT", "") for s in raw_syms_str_norm.split()]
   660→    raw_syms = [s for s in raw_syms if s]  # filter empty
   661→    if not raw_syms:
   662→        raw_syms = ["BTC"]
   663→    timeframes = sort_timeframes(_get_timeframes())
   664→    for i, raw_sym in enumerate(raw_syms):
   665→        symbol = f"{raw_sym}USDT"
   666→        if i > 0:
   667→            await asyncio.sleep(3)  # pause between tickers to avoid rate limit
   668→        await _do_full_scan(symbol, timeframes, message.chat.id, message.bot)
