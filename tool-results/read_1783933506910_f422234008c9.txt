   107→def _query_db(query, args=()):
   108→    try:
   109→        conn = sqlite3.connect(DB_PATH)
   110→        conn.row_factory = sqlite3.Row
   111→        c = conn.cursor()
   112→        c.execute(query, args)
   113→        rows = c.fetchall()
   114→        conn.close()
   115→        return [dict(r) for r in rows]
   116→    except Exception:
   117→        return []

   172→def clean_analysis_report(text: str) -> str:
   173→    """Исправляет логические противоречия в ТП/СЛ и форматирует вывод."""
   174→    if "НАПРАВЛЕНИЕ:" in text:
   175→        direction = "Long" if "Long" in text.split("НАПРАВЛЕНИЕ:")[1].split("|")[0] else "Short"
   176→        price_match = re.search(r"Текущая цена:\s*([\d.]+)", text)
   177→        if price_match:
   178→            current = float(price_match.group(1))
   179→            tp1_match = re.search(r"TP1:\s*([\d.]+)", text)
   180→            if tp1_match:
   181→                tp1 = float(tp1_match.group(1))
   182→                if direction == "Long" and tp1 <= current:
   183→                    text = text.replace(f"TP1: {tp1}", "TP1: Уже отработан")
   184→                elif direction == "Short" and tp1 >= current:
   185→                    text = text.replace(f"TP1: {tp1}", "TP1: Уже отработан")
   186→    return text

   989→# ═══════════════════════════════════════════════════════════════════════════
   990→# АВТОСКАН — последовательный цикл (BTC → 2мин → XAUT → пауза → повтор)
   991→# ═══════════════════════════════════════════════════════════════════════════
   992→
   993→_autoscan_running = False
   994→
   995→
   996→async def _autoscan_sequential_cycle(bot: Bot):
   997→    """Последовательный цикл: символы по очереди, 2 мин пауза между, затем пауза до интервала."""
   998→    global _autoscan_running
   999→    if not DASHBOARD_LLM_API_KEY:
  1000→        logging.warning("autoscan: DASHBOARD_LLM_API_KEY not set")
  1001→        return
  1002→    if _autoscan_running:
  1003→        return
  1004→    _autoscan_running = True
  1005→
  1006→    symbols = _get_autoscan_symbols()
  1007→    interval = _dash_get_setting("autoscan_interval", 30)
  1008→    inter_symbol_pause = 120  # 2 минуты между символами
  1009→
  1010→    import core.config as cfg
  1011→    import core.scheduler as sched_mod
  1012→
  1013→    try:
  1014→        while _dash_get_setting("autoscan_active", False) and _autoscan_running:
  1015→            cycle_start = time.time()
  1016→            timeframes = sort_timeframes(_get_timeframes())
  1017→
  1018→            for i, symbol in enumerate(symbols):
  1019→                if not _dash_get_setting("autoscan_active", False) or not _autoscan_running:
  1020→                    break
  1021→                label = SYMBOL_LABELS.get(symbol, symbol.replace("USDT", ""))
  1022→                logging.info(f"Autoscan: analyzing {label}...")
  1023→
  1024→                # Автоскан: AUTO_SIGNAL_ONLY = True (только сигналы в TG)
  1025→                old_auto = cfg.AUTO_SIGNAL_ONLY
  1026→                old_my_chat = cfg.MY_CHAT_ID
  1027→                try:
  1028→                    cfg.MY_CHAT_ID = ADMIN_CHAT_ID
  1029→                    sched_mod.AUTO_SIGNAL_ONLY = True
  1030→                    sched_mod.ACTIONABLE_SIGNALS = ("aggressive_breakout", "retest", "reversal")
  1031→                    from core.scheduler import run_hourly_analysis
  1032→                    await run_hourly_analysis(
  1033→                        bot=bot,
  1034→                        symbol_filter=symbol,
  1035→                        llm_api_key=DASHBOARD_LLM_API_KEY,
  1036→                        llm_base_url=DASHBOARD_LLM_BASE_URL,
  1037→                        llm_model=DASHBOARD_MODEL_NAME,
  1038→                    )
  1039→                except Exception as e:
  1040→                    logging.error(f"autoscan {symbol} error: {e}")
  1041→                finally:
  1042→                    cfg.AUTO_SIGNAL_ONLY = old_auto
  1043→                    cfg.MY_CHAT_ID = old_my_chat
  1044→
  1045→                # Пауза между символами
  1046→                if i < len(symbols) - 1 and _dash_get_setting("autoscan_active", False):
  1047→                    logging.info(f"Autoscan: pause {inter_symbol_pause}s before next symbol")
  1048→                    sleep_end = time.time() + inter_symbol_pause
  1049→                    while time.time() < sleep_end and _autoscan_running:
  1050→                        if not _dash_get_setting("autoscan_active", False):
  1051→                            break
  1052→                        await asyncio.sleep(min(10, sleep_end - time.time()))
  1053→
  1054→            # Пауза после полного цикла
  1055→            if _dash_get_setting("autoscan_active", False) and _autoscan_running:
  1056→                elapsed = time.time() - cycle_start
  1057→                remaining = (interval * 60) - elapsed
  1058→                if remaining > 0:
  1059→                    logging.info(f"Autoscan: cycle done, waiting {remaining:.0f}s")
  1060→                    sleep_end = time.time() + remaining
  1061→                    while time.time() < sleep_end and _autoscan_running:
  1062→                        if not _dash_get_setting("autoscan_active", False):
  1063→                            break
  1064→                        await asyncio.sleep(min(10, sleep_end - time.time()))
  1065→    except asyncio.CancelledError:
  1066→        logging.info("Autoscan: cancelled")
  1067→    finally:
  1068→        _autoscan_running = False
  1069→
  1070→
  1071→def _start_autoscan(bot: Bot) -> bool:
  1072→    global _autoscan_running
  1073→    if _autoscan_running:
  1074→        return False
  1075→    _dash_set_setting("autoscan_active", True)
  1076→    symbols = _get_autoscan_symbols()
  1077→    interval = _dash_get_setting("autoscan_interval", 30)
  1078→    labels = ", ".join(SYMBOL_LABELS.get(s, s.replace("USDT", "")) for s in symbols)
  1079→    logging.info(f"Autoscan started: interval={interval}min, symbols=[{labels}]")
  1080→    asyncio.create_task(_autoscan_sequential_cycle(bot))
  1081→    return True
  1082→

  1084→def _stop_autoscan() -> bool:
  1085→    global _autoscan_running
  1086→    _autoscan_running = False
  1087→    _dash_set_setting("autoscan_active", False)
  1088→    logging.info("Autoscan stopped")
