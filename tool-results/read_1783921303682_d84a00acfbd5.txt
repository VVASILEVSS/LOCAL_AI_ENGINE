    59→def start_main_bot() -> bool:

    61→    if main_bot_process and main_bot_process.poll() is None:
    62→        return False

    73→def stop_main_bot() -> bool:
    74→    global main_bot_process
    75→    if not main_bot_process or main_bot_process.poll() is not None:
    76→        return False
    77→    main_bot_process.terminate()
    78→    try:
    79→        main_bot_process.wait(timeout=10)
    80→    except subprocess.TimeoutExpired:

    89→def _query_db(query, args=()):
    90→    try:
    91→        conn = sqlite3.connect(DB_PATH)
    92→        conn.row_factory = sqlite3.Row
    93→        c = conn.cursor()
    94→        c.execute(query, args)
    95→        rows = c.fetchall()
    96→        conn.close()
    97→        return [dict(r) for r in rows]
    98→    except Exception:
    99→        return []

   101→def get_setting(key, default=None):
   102→    rows = _query_db("SELECT value FROM settings WHERE key=?", (key,))
   103→    if rows:
   104→        v = rows[0]["value"]
   105→        if v in ("true", "false"): return v == "true"
   106→        return v
   107→    return default
   108→
   109→def set_setting(key, value):
   110→    conn = sqlite3.connect(DB_PATH)
   111→    c = conn.cursor()
   112→    c.execute("INSERT OR REPLACE INTO settings (key, value) VALUES (?, ?)", (key, str(value)))
   113→    conn.commit(); conn.close()

   115→def get_stats():
   116→    stats = {
   117→        "bot_status": "🟢 Running" if is_main_bot_running() else "🔴 Stopped",
   118→        "uptime": int(time.time() - main_bot_started_at) if main_bot_started_at else 0,
   119→        "model": DASHBOARD_MODEL_NAME,
   120→        "llm_mode": "cloud" if DASHBOARD_LLM_API_KEY else "local",
   121→        "prompt_variant": os.getenv("PROMPT_VARIANT", "A"),
   122→        "interval": get_setting("interval_minutes", 60),
   123→        "auto_mode": get_setting("auto_mode", False),
   124→    }
   125→    rows = _query_db("""
   126→        SELECT COUNT(*) as total,
   127→            SUM(CASE WHEN outcome IN ('tp1_hit','tp2_hit','tp3_hit') THEN 1 ELSE 0 END) as wins,
   128→            SUM(CASE WHEN outcome = 'sl_hit' THEN 1 ELSE 0 END) as sl_hits
   129→        FROM signal_log WHERE checked_at IS NOT NULL
   130→    """)
   131→    if rows and rows[0]["total"] and rows[0]["total"] > 0:
   132→        r = rows[0]; total = r["total"]
   133→        stats.update(checked=total, wins=int(r["wins"] or 0),
   134→                      accuracy=round((r["wins"] or 0) / total * 100, 1),
   135→                      sl_rate=round((r["sl_hits"] or 0) / total * 100, 1))
   136→    else:
   137→        stats.update(checked=0, wins=0, accuracy=0, sl_rate=0)
   138→    pending = _query_db("SELECT COUNT(*) as cnt FROM signal_log WHERE checked_at IS NULL")
   139→    stats["pending"] = pending[0]["cnt"] if pending else 0
   140→    last = _query_db("SELECT timestamp, symbol, signal_status, direction, entry_price, sl, tp1, outcome FROM signal_log ORDER BY timestamp DESC LIMIT 10")
   141→    stats["last_signals"] = last
   142→    ab = _query_db("SELECT prompt_variant, COUNT(*) as cnt, SUM(CASE WHEN outcome IN ('tp1_hit','tp2_hit','tp3_hit') THEN 1 ELSE 0 END) as wins FROM signal_log WHERE checked_at IS NOT NULL AND prompt_variant IS NOT NULL GROUP BY prompt_variant")
   143→    stats["ab_variants"] = [{"variant": r["prompt_variant"] or "A", "total": r["cnt"], "wins": r["wins"] or 0} for r in ab]
   144→    return stats

   455→async def _autoscan_sequential_cycle(bot: Bot):
   456→    """Последовательный цикл: BTC → пауза 2 мин → XAUT → пауза (interval - 2 мин) → повтор.
   457→    
   458→    Логика из письма Гермеса: сканирует выбранные символы по очереди,
   459→    между символами пауза 2 мин, после последнего — пауза до полного интервала.
   460→    """
   461→    global _autoscan_running
   462→    if not DASHBOARD_LLM_API_KEY:
   463→        logging.warning("autoscan: DASHBOARD_LLM_API_KEY not set, skipping")
   464→        return
   465→    if _autoscan_running:
   466→        return
   467→    _autoscan_running = True
   468→
   469→    symbols = _get_autoscan_symbols()
   470→    interval = get_setting("autoscan_interval", 30)
   471→    # Пауза между символами внутри цикла
   472→    inter_symbol_pause = 120  # 2 минуты
   473→    # Если интервал меньше суммы пауз — уменьшаем
   474→    if len(symbols) > 1:
   475→        total_internal_pause = inter_symbol_pause * (len(symbols) - 1)
   476→    else:
   477→        total_internal_pause = 0
   478→
   479→    import core.config as cfg
   480→    import core.scheduler as sched_mod
   481→
   482→    try:
   483→        while get_setting("autoscan_active", False) and _autoscan_running:
   484→            cycle_start = time.time()
   485→            for i, symbol in enumerate(symbols):
   486→                if not get_setting("autoscan_active", False) or not _autoscan_running:
   487→                    break
   488→                label = SYMBOL_LABELS.get(symbol, symbol.replace("USDT", ""))
   489→                logging.info(f"Autoscan: analyzing {label}...")
   490→                old_auto = cfg.AUTO_SIGNAL_ONLY
   491→                old_my_chat = cfg.MY_CHAT_ID
   492→                try:
   493→                    cfg.MY_CHAT_ID = ADMIN_CHAT_ID
   494→                    sched_mod.AUTO_SIGNAL_ONLY = True
   495→                    sched_mod.ACTIONABLE_SIGNALS = ("aggressive_breakout", "retest", "reversal")
   496→                    await run_hourly_analysis(
   497→                        bot=bot,
   498→                        symbol_filter=symbol,
   499→                        llm_api_key=DASHBOARD_LLM_API_KEY,
   500→                        llm_base_url=DASHBOARD_LLM_BASE_URL,
   501→                        llm_model=DASHBOARD_MODEL_NAME,
   502→                    )
   503→                except Exception as e:
   504→                    logging.error(f"autoscan {symbol} error: {e}")
   505→                finally:
   506→                    cfg.AUTO_SIGNAL_ONLY = old_auto
   507→                    cfg.MY_CHAT_ID = old_my_chat
   508→
   509→                # Пауза между символами (кроме последнего)
   510→                if i < len(symbols) - 1 and get_setting("autoscan_active", False):
   511→                    logging.info(f"Autoscan: pause {inter_symbol_pause}s before next symbol")
   512→                    await asyncio.sleep(inter_symbol_pause)
   513→
   514→            # Пауза после полного цикла до наступления следующего интервала
   515→            if get_setting("autoscan_active", False) and _autoscan_running:
   516→                elapsed = time.time() - cycle_start
   517→                remaining = (interval * 60) - elapsed
   518→                if remaining > 0:
   519→                    logging.info(f"Autoscan: cycle done, waiting {remaining:.0f}s for next cycle")
   520→                    # Спим короткими интервалами, чтобы можно было остановить
   521→                    sleep_end = time.time() + remaining
   522→                    while time.time() < sleep_end:
   523→                        if not get_setting("autoscan_active", False) or not _autoscan_running:
   524→                            break
   525→                        await asyncio.sleep(min(10, sleep_end - time.time()))
   526→    except asyncio.CancelledError:
   527→        logging.info("Autoscan: cancelled")
   528→    finally:
   529→        _autoscan_running = False
   530→
   531→
   532→def _start_autoscan(bot: Bot) -> bool:
   533→    global _autoscan_running
   534→    if _autoscan_running:
   535→        return False
   536→    set_setting("autoscan_active", True)
   537→    symbols = _get_autoscan_symbols()
   538→    interval = get_setting("autoscan_interval", 30)
   539→    labels = ", ".join(SYMBOL_LABELS.get(s, s.replace("USDT", "")) for s in symbols)
   540→    logging.info(f"Autoscan started: interval={interval}min, symbols=[{labels}]")
   541→    # Запускаем цикл как задачу (не через APScheduler interval — управляем паузами вручную)
   542→    asyncio.create_task(_autoscan_sequential_cycle(bot))
   543→    return True
