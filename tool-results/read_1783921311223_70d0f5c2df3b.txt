    89→    89→def _query_db(query, args=()):
    90→    90→    try:
    91→    91→        conn = sqlite3.connect(DB_PATH)
    92→    92→        conn.row_factory = sqlite3.Row
    93→    93→        c = conn.cursor()
    94→    94→        c.execute(query, args)
    95→    95→        rows = c.fetchall()
    96→    96→        conn.close()
    97→    97→        return [dict(r) for r in rows]
    98→    98→    except Exception:
    99→    99→        return []

   115→   115→def get_stats():
   116→   116→    stats = {
   117→   117→        "bot_status": "🟢 Running" if is_main_bot_running() else "🔴 Stopped",
   118→   118→        "uptime": int(time.time() - main_bot_started_at) if main_bot_started_at else 0,

   121→   121→        "prompt_variant": os.getenv("PROMPT_VARIANT", "A"),
   122→   122→        "interval": get_setting("interval_minutes", 60),
   123→   123→        "auto_mode": get_setting("auto_mode", False),
   124→   124→    }
   125→   125→    rows = _query_db("""
   126→   126→        SELECT COUNT(*) as total,
   127→   127→            SUM(CASE WHEN outcome IN ('tp1_hit','tp2_hit','tp3_hit') THEN 1 ELSE 0 END) as wins,
   128→   128→            SUM(CASE WHEN outcome = 'sl_hit' THEN 1 ELSE 0 END) as sl_hits
   129→   129→        FROM signal_log WHERE checked_at IS NOT NULL
   130→   130→    """)
   131→   131→    if rows and rows[0]["total"] and rows[0]["total"] > 0:
   132→   132→        r = rows[0]; total = r["total"]
   133→   133→        stats.update(checked=total, wins=int(r["wins"] or 0),
   134→   134→                      accuracy=round((r["wins"] or 0) / total * 100, 1),
   135→   135→                      sl_rate=round((r["sl_hits"] or 0) / total * 100, 1))
   136→   136→    else:
   137→   137→        stats.update(checked=0, wins=0, accuracy=0, sl_rate=0)
   138→   138→    pending = _query_db("SELECT COUNT(*) as cnt FROM signal_log WHERE checked_at IS NULL")
   139→   139→    stats["pending"] = pending[0]["cnt"] if pending else 0
   140→   140→    last = _query_db("SELECT timestamp, symbol, signal_status, direction, entry_price, sl, tp1, outcome FROM signal_log ORDER BY timestamp DESC LIMIT 10")
   141→   141→    stats["last_signals"] = last
   142→   142→    ab = _query_db("SELECT prompt_variant, COUNT(*) as cnt, SUM(CASE WHEN outcome IN ('tp1_hit','tp2_hit','tp3_hit') THEN 1 ELSE 0 END) as wins FROM signal_log WHERE checked_at IS NOT NULL AND prompt_variant IS NOT NULL GROUP BY prompt_variant")
   143→   143→    stats["ab_variants"] = [{"variant": r["prompt_variant"] or "A", "total": r["cnt"], "wins": r["wins"] or 0} for r in ab]
   144→   144→    return stats

   454→   454→
   455→   455→async def _autoscan_sequential_cycle(bot: Bot):
   456→   456→    """Последовательный цикл: BTC → пауза 2 мин → XAUT → пауза (interval - 2 мин) → повтор.
   457→   457→    
   458→   458→    Логика из письма Гермеса: сканирует выбранные символы по очереди,
   459→   459→    между символами пауза 2 мин, после последнего — пауза до полного интервала.
   460→   460→    """
   461→   461→    global _autoscan_running
   462→   462→    if not DASHBOARD_LLM_API_KEY:
   463→   463→        logging.warning("autoscan: DASHBOARD_LLM_API_KEY not set, skipping")
   464→   464→        return
   465→   465→    if _autoscan_running:
   466→   466→        return
   467→   467→    _autoscan_running = True
   468→   468→
   469→   469→    symbols = _get_autoscan_symbols()
   470→   470→    interval = get_setting("autoscan_interval", 30)
   471→   471→    # Пауза между символами внутри цикла
   472→   472→    inter_symbol_pause = 120  # 2 минуты
   473→   473→    # Если интервал меньше суммы пауз — уменьшаем
   474→   474→    if len(symbols) > 1:
   475→   475→        total_internal_pause = inter_symbol_pause * (len(symbols) - 1)
   476→   476→    else:
   477→   477→        total_internal_pause = 0
   478→   478→
   479→   479→    import core.config as cfg
   480→   480→    import core.scheduler as sched_mod
   481→   481→
   482→   482→    try:
   483→   483→        while get_setting("autoscan_active", False) and _autoscan_running:
   484→   484→            cycle_start = time.time()
   485→   485→            for i, symbol in enumerate(symbols):
   486→   486→                if not get_setting("autoscan_active", False) or not _autoscan_running:
   487→   487→                    break
   488→   488→                label = SYMBOL_LABELS.get(symbol, symbol.replace("USDT", ""))
   489→   489→                logging.info(f"Autoscan: analyzing {label}...")
   490→   490→                old_auto = cfg.AUTO_SIGNAL_ONLY
   491→   491→                old_my_chat = cfg.MY_CHAT_ID
   492→   492→                try:
   493→   493→                    cfg.MY_CHAT_ID = ADMIN_CHAT_ID
   494→   494→                    sched_mod.AUTO_SIGNAL_ONLY = True
   495→   495→                    sched_mod.ACTIONABLE_SIGNALS = ("aggressive_breakout", "retest", "reversal")
   496→   496→                    await run_hourly_analysis(
   497→   497→                        bot=bot,
   498→   498→                        symbol_filter=symbol,
   499→   499→                        llm_api_key=DASHBOARD_LLM_API_KEY,
   500→   500→                        llm_base_url=DASHBOARD_LLM_BASE_URL,
   501→   501→                        llm_model=DASHBOARD_MODEL_NAME,
   502→   502→                    )
   503→   503→                except Exception as e:
   504→   504→                    logging.error(f"autoscan {symbol} error: {e}")
   505→   505→                finally:
   506→   506→                    cfg.AUTO_SIGNAL_ONLY = old_auto
   507→   507→                    cfg.MY_CHAT_ID = old_my_chat
   508→   508→
   509→   509→                # Пауза между символами (кроме последнего)
   510→   510→                if i < len(symbols) - 1 and get_setting("autoscan_active", False):
   511→   511→                    logging.info(f"Autoscan: pause {inter_symbol_pause}s before next symbol")
   512→   512→                    await asyncio.sleep(inter_symbol_pause)
   513→   513→
   514→   514→            # Пауза после полного цикла до наступления следующего интервала
   515→   515→            if get_setting("autoscan_active", False) and _autoscan_running:
   516→   516→                elapsed = time.time() - cycle_start
   517→   517→                remaining = (interval * 60) - elapsed
   518→   518→                if remaining > 0:
   519→   519→                    logging.info(f"Autoscan: cycle done, waiting {remaining:.0f}s for next cycle")
   520→   520→                    # Спим короткими интервалами, чтобы можно было остановить
   521→   521→                    sleep_end = time.time() + remaining
   522→   522→                    while time.time() < sleep_end:
   523→   523→                        if not get_setting("autoscan_active", False) or not _autoscan_running:
   524→   524→                            break
   525→   525→                        await asyncio.sleep(min(10, sleep_end - time.time()))
   526→   526→    except asyncio.CancelledError:
   527→   527→        logging.info("Autoscan: cancelled")
   528→   528→    finally:
   529→   529→        _autoscan_running = False

   532→   532→def _start_autoscan(bot: Bot) -> bool:
   533→   533→    global _autoscan_running
   534→   534→    if _autoscan_running:
   535→   535→        return False
   536→   536→    set_setting("autoscan_active", True)
   537→   537→    symbols = _get_autoscan_symbols()

   541→   541→    # Запускаем цикл как задачу (не через APScheduler interval — управляем паузами вручную)
   542→   542→    asyncio.create_task(_autoscan_sequential_cycle(bot))
