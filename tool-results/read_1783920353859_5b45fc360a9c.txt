    89→def _query_db(query, args=()):
    90→    try:
    91→        conn = sqlite3.connect(DB_PATH)
    92→        conn.row_factory = sqlite3.Row
    93→        c = conn.cursor()
    94→        c.execute(query, args)
    95→        rows = c.fetchall()
    96→        conn.close()
    97→        return [dict(r) for r in rows]
    98→    except Exception:
    99→        return []

   115→def get_stats():
   116→    stats = {
   117→        "bot_status": "🟢 Running" if is_main_bot_running() else "🔴 Stopped",
   118→        "uptime": int(time.time() - main_bot_started_at) if main_bot_started_at else 0,
   119→        "model": DASHBOARD_MODEL_NAME,
   120→        "llm_mode": "cloud" if DASHBOARD_LLM_API_KEY else "local",
   121→        "prompt_variant": os.getenv("PROMPT_VARIANT", "A"),
   122→        "interval": get_setting("interval_minutes", 60),
   123→        "auto_mode": get_setting("auto_mode", False),
   124→    }
   125→    rows = _query_db("""
   126→        SELECT COUNT(*) as total,
   127→            SUM(CASE WHEN outcome IN ('tp1_hit','tp2_hit','tp3_hit') THEN 1 ELSE 0 END) as wins,
   128→            SUM(CASE WHEN outcome = 'sl_hit' THEN 1 ELSE 0 END) as sl_hits
   129→        FROM signal_log WHERE checked_at IS NOT NULL
   130→    """)
   131→    if rows and rows[0]["total"] and rows[0]["total"] > 0:
   132→        r = rows[0]; total = r["total"]
   133→        stats.update(checked=total, wins=int(r["wins"] or 0),
   134→                      accuracy=round((r["wins"] or 0) / total * 100, 1),
   135→                      sl_rate=round((r["sl_hits"] or 0) / total * 100, 1))
   136→    else:
   137→        stats.update(checked=0, wins=0, accuracy=0, sl_rate=0)
   138→    pending = _query_db("SELECT COUNT(*) as cnt FROM signal_log WHERE checked_at IS NULL")
   139→    stats["pending"] = pending[0]["cnt"] if pending else 0
   140→    last = _query_db("SELECT timestamp, symbol, signal_status, direction, entry_price, sl, tp1, outcome FROM signal_log ORDER BY timestamp DESC LIMIT 10")
   141→    stats["last_signals"] = last
   142→    ab = _query_db("SELECT prompt_variant, COUNT(*) as cnt, SUM(CASE WHEN outcome IN ('tp1_hit','tp2_hit','tp3_hit') THEN 1 ELSE 0 END) as wins FROM signal_log WHERE checked_at IS NOT NULL AND prompt_variant IS NOT NULL GROUP BY prompt_variant")
   143→    stats["ab_variants"] = [{"variant": r["prompt_variant"] or "A", "total": r["cnt"], "wins": r["wins"] or 0} for r in ab]
   144→    return stats

   289→        # Автоскан: AUTO_SIGNAL_ONLY — только сигналы
   290→        cfg.MY_CHAT_ID = ADMIN_CHAT_ID
   291→        sched_mod.AUTO_SIGNAL_ONLY = True

   305→
   306→def _start_autoscan(bot: Bot) -> bool:
   307→    interval = get_setting("autoscan_interval", 30)
   308→    autoscan_scheduler.add_job(
   309→        _autoscan_cycle, "interval", minutes=interval,
   310→        args=[bot], id="autoscan_job", replace_existing=True,
   311→        max_instances=1, coalesce=True,
   312→    )
   313→    if not autoscan_scheduler.running:
   314→        autoscan_scheduler.start()
   315→    set_setting("autoscan_active", True)
   316→    logging.info(f"Autoscan started: every {interval} min, symbols: {SYMBOLS_SCAN}")
   317→    return True

   478→    if not text:
   479→        await message.answer("Использование: /scan BTC | ETH | XAUT", reply_markup=_main_keyboard())

   482→    symbol = text.upper().split()[0]
   483→    full = symbol_map.get(symbol, symbol + "USDT" if not symbol.endswith("USDT") else symbol)
   484→    await message.answer(f"🔍 Анализ {full} через облако ({DASHBOARD_MODEL_NAME})...")

   488→async def _do_scan(bot: Bot, symbol: str, chat_id: int):
   489→    """Общий код сканирования через облако (для /scan и кнопок)."""
   490→    try:
   491→        from core.scheduler import run_hourly_analysis
   492→        import core.config as cfg
   493→        import core.scheduler as sched_mod
   494→        # Сохраняем оригинальные значения
   495→        old_chat = cfg.MY_CHAT_ID
   496→        old_auto = cfg.AUTO_SIGNAL_ONLY
   497→        try:
   498→            cfg.MY_CHAT_ID = chat_id
   499→            # Ручной /scan ВСЕГДА отправляет результат
   500→            cfg.AUTO_SIGNAL_ONLY = False
   501→            sched_mod.AUTO_SIGNAL_ONLY = False
   502→            await run_hourly_analysis(
   503→                bot=bot,
   504→                symbol_filter=symbol,
   505→                llm_api_key=DASHBOARD_LLM_API_KEY,
   506→                llm_base_url=DASHBOARD_LLM_BASE_URL,
   507→                llm_model=DASHBOARD_MODEL_NAME,
   508→            )
   509→        finally:
   510→            cfg.MY_CHAT_ID = old_chat
   511→            cfg.AUTO_SIGNAL_ONLY = old_auto
   512→            sched_mod.AUTO_SIGNAL_ONLY = old_auto
