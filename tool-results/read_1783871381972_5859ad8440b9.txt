   335→def prepare_image_for_llm(img_bytes: bytes) -> str:
   336→    try:
   337→        img = Image.open(io.BytesIO(img_bytes))
   338→        if img.mode != "RGB":
   339→            img = img.convert("RGB")
   340→        img.thumbnail((1024, 1024))
   341→        buffer = io.BytesIO()
   342→        img.save(buffer, format="JPEG", quality=85)
   343→        return base64.b64encode(buffer.getvalue()).decode("utf-8")
   344→    except Exception as e:
   345→        logger.error(f"Ошибка подготовки фото: {e}")
   346→        return ""

   349→def _fmt(v: Any) -> str:
   350→    if v is None or v == "None" or v == "":
   351→        return "Н/Д"
   352→    return str(v)

   355→def parse_llm_json(raw: str) -> dict:
   356→    clean = raw.strip()
   357→    clean = re.sub(r"^```(?:json)?\s*", "", clean)
   358→    clean = re.sub(r"\s*```$", "", clean)
   359→
   360→    start = clean.find("{")
   361→    end = clean.rfind("}")
   362→    if start == -1 or end == -1 or end <= start:
   363→        return {"error": True, "message": "JSON not found", "raw": raw}
   364→
   365→    candidate = clean[start:end + 1].strip()
   366→    candidate = candidate.replace("“", '"').replace("”", '"').replace("’", "'")
   367→
   368→    # Базовая очистка
   369→    candidate = re.sub(r'("\s*)(\r?\n\s*")', r'\1,\2', candidate)
   370→    candidate = re.sub(r'([}\]])(\r?\n\s*")', r'\1,\2', candidate)
   371→    candidate = re.sub(r',\s*([}\]])', r'\1', candidate)

   437→            rm = data["risk_management"]
   438→
   439→            if "primary" not in rm and any(k in rm for k in ("sl", "tp1", "tp2", "tp3", "rr")):
   440→                rm = {

  1747→def _merge_close_levels(levels: list[float], tolerance_ratio: float = 0.0015) -> list[float]:
  1748→    if not levels:
  1749→        return []
  1750→    levels = sorted(float(x) for x in levels if x is not None)
  1751→    merged = [levels[0]]
  1752→    for lvl in levels[1:]:
  1753→        last = merged[-1]
  1754→        tol = max(abs(last) * tolerance_ratio, 0.5)
  1755→        if abs(lvl - last) <= tol:
  1756→            merged[-1] = round((last + lvl) / 2.0, 6)
  1757→        else:
  1758→            merged.append(lvl)
  1759→    return merged
