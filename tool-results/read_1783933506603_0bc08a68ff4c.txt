   335→def prepare_image_for_llm(img_bytes: bytes) -> str:
   336→    try:
   337→        img = Image.open(io.BytesIO(img_bytes))
   338→        if img.mode != "RGB":
   339→            img = img.convert("RGB")
   340→        img.thumbnail((1024, 1024))
   341→        buffer = io.BytesIO()
   342→        img.save(buffer, format="JPEG", quality=85)
   343→        return base64.b64encode(buffer.getvalue()).decode("utf-8")
   344→    except Exception as e:
   345→        logger.error(f"Ошибка подготовки фото: {e}")
   346→        return ""

   349→def _fmt(v: Any) -> str:
   350→    if v is None or v == "None" or v == "":
   351→        return "Н/Д"
   352→    return str(v)

   355→def parse_llm_json(raw: str) -> dict:
   356→    clean = raw.strip()
   357→    clean = re.sub(r"^```(?:json)?\s*", "", clean)
   358→    clean = re.sub(r"\s*```$", "", clean)
   359→
   360→    start = clean.find("{")
   361→    end = clean.rfind("}")
   362→    if start == -1 or end == -1 or end <= start:
   363→        return {"error": True, "message": "JSON not found", "raw": raw}
   364→
   365→    candidate = clean[start:end + 1].strip()
   366→    candidate = candidate.replace("“", '"').replace("”", '"').replace("’", "'")
   367→
   368→    # Базовая очистка
   369→    candidate = re.sub(r'("\s*)(\r?\n\s*")', r'\1,\2', candidate)
   370→    candidate = re.sub(r'([}\]])(\r?\n\s*")', r'\1,\2', candidate)
   371→    candidate = re.sub(r',\s*([}\]])', r'\1', candidate)

   876→    def _validate_zone_nesting(tf_zones: dict, price: float | None) -> dict:
   877→        """
   878→        1. Младший ТФ должен быть ВНУТРИ старшего: lower_child >= lower_parent,
   879→           upper_child <= upper_parent. Если нарушено — расширяем родительскую зону.
   880→        2. D1 зона ограничивается ±10% от текущей цены (LLM иногда берёт
   881→           исторический максимум за 100 свечей).
   882→        Порядок вложенности: 1D ⊃ 4H ⊃ 1H ⊃ 15M ⊃ 5M
   883→        """
   884→        if not tf_zones or price is None:
   885→            return tf_zones
   886→
   887→        nesting_order = ["1D", "4H", "1H", "15M", "5M"]
   888→        cap_pct = 0.10  # ±10% от цены
   889→
   890→        # 2a-1: ограничить D1 зону
   891→        d1 = tf_zones.get("1D")
   892→        if isinstance(d1, dict):
   893→            d1_upper = d1.get("upper")
   894→            d1_lower = d1.get("lower")
   895→            if d1_upper is not None and abs(d1_upper - price) / price > cap_pct:
   896→                d1["upper"] = round(price * (1 + cap_pct), 2)
   897→            if d1_lower is not None and abs(price - d1_lower) / price > cap_pct:
