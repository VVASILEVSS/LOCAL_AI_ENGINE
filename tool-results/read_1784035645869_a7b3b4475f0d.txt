   338→def prepare_image_for_llm(img_bytes: bytes) -> str:
   339→    try:
   340→        img = Image.open(io.BytesIO(img_bytes))
   341→        if img.mode != "RGB":
   342→            img = img.convert("RGB")
   343→        img.thumbnail((1024, 1024))
   344→        buffer = io.BytesIO()
   345→        img.save(buffer, format="JPEG", quality=85)
   346→        return base64.b64encode(buffer.getvalue()).decode("utf-8")
   347→    except Exception as e:

   358→def parse_llm_json(raw: str) -> dict:
   359→    clean = raw.strip()
   360→    clean = re.sub(r"^```(?:json)?\s*", "", clean)
   361→    clean = re.sub(r"\s*```$", "", clean)
   362→
   363→    start = clean.find("{")
   364→    end = clean.rfind("}")
   365→    if start == -1 or end == -1 or end <= start:
   366→        return {"error": True, "message": "JSON not found", "raw": raw}
   367→
   368→    candidate = clean[start:end + 1].strip()
   369→    candidate = candidate.replace("“", '"').replace("”", '"').replace("’", "'")
   370→
   371→    # Базовая очистка
   372→    candidate = re.sub(r'("\s*)(\r?\n\s*")', r'\1,\2', candidate)
   373→    candidate = re.sub(r'([}\]])(\r?\n\s*")', r'\1,\2', candidate)
   374→    candidate = re.sub(r',\s*([}\]])', r'\1', candidate)
   375→
   376→    # Убираем двойные запятые и лишние разделители
   377→    candidate = re.sub(r',\s*,+', ',', candidate)
   378→    candidate = re.sub(r'([}\]])\s*,\s*([}\]])', r'\1\2', candidate)
   379→    candidate = re.sub(r'(".*?")\s*,\s*,\s*\n', r'\1,\n', candidate)
   380→    candidate = re.sub(r'(".*?")\s*,\s*,', r'\1,', candidate)
   381→
   382→    try:
   383→        data = json.loads(candidate)
   384→
   385→        # Нормализация числовых полей верхнего уровня
   386→        for key in ("price", "current_price", "last_closed_price"):
   387→            if key in data:
   388→                data[key] = _safe_float(data.get(key))
   389→
   390→        # Нормализация tf_zones
   391→        if isinstance(data.get("tf_zones"), dict):
   392→            normalized = {}
   393→            for k, v in data["tf_zones"].items():
   394→                tf_key = str(k).strip().upper().replace("MIN", "M")
   395→                if isinstance(v, dict):
   396→                    normalized[tf_key] = {
   397→                        "upper": _safe_float(v.get("upper")),
   398→                        "lower": _safe_float(v.get("lower")),
   399→                    }
   400→                else:
   401→                    normalized[tf_key] = v
   402→            data["tf_zones"] = normalized
   403→
   404→        # Нормализация key_zones
   405→        if isinstance(data.get("key_zones"), dict):
   406→            kz = data["key_zones"]
   407→            for k in ("resistance", "support"):
   408→                v = kz.get(k)
   409→                if isinstance(v, list):
   410→                    num = next((x for x in v if isinstance(x, (int, float))), None)
   411→                    kz[k] = float(num) if num is not None else None
   412→                else:
   413→                    kz[k] = _safe_float(v)
   414→
   415→        # Нормализация tf_span_map
   416→        if isinstance(data.get("tf_span_map"), dict):
   417→            span_map = {}
   418→            for k, v in data["tf_span_map"].items():
   419→                span_map[str(k).strip().upper()] = _safe_float(v)
   420→            data["tf_span_map"] = span_map
   421→
   422→        # Нормализация confluence_levels
   423→        if isinstance(data.get("confluence_levels"), list):
   424→            clean_levels = []
   425→            for item in data["confluence_levels"]:
   426→                if not isinstance(item, dict):
   427→                    continue
   428→                clean_levels.append({
   429→                    "level": _safe_float(item.get("level")),
   430→                    "timeframes": [str(x).upper() for x in item.get("timeframes", []) if str(x).strip()],
   431→                    "priority": str(item.get("priority", "low")).lower(),
   432→                    "count": int(item.get("count") or 0),
   433→                    "spread": _safe_float(item.get("spread")),
   434→                    "kind": str(item.get("kind", "mixed")).lower(),
   435→                })
   436→            data["confluence_levels"] = clean_levels
   437→
   438→        # Нормализация risk_management
   439→        if isinstance(data.get("risk_management"), dict):
   440→            rm = data["risk_management"]
   441→
   442→            if "primary" not in rm and any(k in rm for k in ("sl", "tp1", "tp2", "tp3", "rr")):
   443→                rm = {
   444→                    "primary": {
   445→                        "sl": rm.get("sl"),
   446→                        "tp1": rm.get("tp1"),
   447→                        "tp2": rm.get("tp2"),
   448→                        "tp3": rm.get("tp3"),
   449→                        "rr": rm.get("rr"),
   450→                    },
   451→                    "alternative": {
   452→                        "sl": None,
   453→                        "tp1": None,
   454→                        "tp2": None,
   455→                        "tp3": None,
   456→                        "rr": None,
   457→                    },
   458→                }
   459→
   460→            for branch in ("primary", "alternative"):
   461→                if branch not in rm or not isinstance(rm[branch], dict):
   462→                    rm[branch] = {
   463→                        "sl": None,
   464→                        "tp1": None,
   465→                        "tp2": None,
   466→                        "tp3": None,
   467→                        "rr": None,
   468→                    }
   469→                else:
   470→                    for k in ("sl", "tp1", "tp2", "tp3", "rr"):
   471→                        rm[branch][k] = _safe_float(rm[branch].get(k))
   472→
   473→            data["risk_management"] = rm
   474→
   475→        return data
   476→
   477→    except json.JSONDecodeError as e:
   478→        logger.warning(f"LLM parse failed after cleanup: {e}")
   479→        return {
   480→            "error": True,
   481→            "message": f"Parse failed: {str(e)}",
   482→            "raw": raw,
   483→            "candidate": candidate[:4000],
   484→        }

   681→    def _normalize_confluence_levels(levels) -> list:
   682→        if not isinstance(levels, list):
   683→            return []
   684→
   685→        out = []
   686→        for item in levels:
   687→            if not isinstance(item, dict):
   688→                continue
   689→            timeframes = item.get("timeframes", [])
   690→            if not isinstance(timeframes, list):
   691→                timeframes = []
   692→            lvl = _safe_float(item.get("level"))
   693→            out.append({
   694→                "level": lvl,
   695→                "timeframes": [str(tf).upper() for tf in timeframes if str(tf).strip()],
   696→                "priority": str(item.get("priority", "low")).lower(),
   697→                "count": int(item.get("count") or 0),
   698→                "spread": _safe_float(item.get("spread")),
   699→                "kind": str(item.get("kind", "mixed")).lower(),
   700→            })
   701→
   702→        # FEELS-inspired: пересчитываем spread через log-distance
   703→        # |ln(level/price)| — симметричен: +50% и -50% дают ~0.405
   704→        # Это решает проблему D1 support на -30% получая штраф в $
   705→        # а ближний resistance на +2% в $ почти не штрафуется.
   706→        price_for_ld = _safe_float(data.get("price") or data.get("current_price"))
   707→        if price_for_ld and price_for_ld > 0:
   708→            import math
   709→            for item in out:
   710→                lvl = item.get("level")
   711→                if lvl and lvl > 0:
   712→                    item["log_distance"] = round(abs(math.log(lvl / price_for_ld)), 6)
   713→                    # proximity_score: инвертированный-U, пик ~2% (log_dist≈0.02)
   714→                    ld = item["log_distance"]
   715→                    if ld < 0.002:
   716→                        item["proximity_score"] = 0.3
   717→                    elif ld < 0.05:
   718→                        item["proximity_score"] = round(max(0.5, 1.0 - abs(ld - 0.02) * 10.0), 4)
   719→                    else:
   720→                        item["proximity_score"] = round(max(0.05, 1.0 - (ld - 0.05) * 3.0), 4)
   721→                else:
   722→                    item["log_distance"] = None
   723→                    item["proximity_score"] = None
   724→
   725→        return out

   902→    # -----------------------------
   903→    # 2a) Валидация матрёшки зон + ограничение D1
   904→    # -----------------------------
   905→    def _validate_zone_nesting(tf_zones: dict, price: float | None) -> dict:
   906→        """
   907→        1. Младший ТФ должен быть ВНУТРИ старшего: lower_child >= lower_parent,
   908→           upper_child <= upper_parent. Если нарушено — СУЖАЕМ CHILD до parent,
   909→           потому что старший ТФ авторитетнее (D1 ⊃ 4H ⊃ 1H ⊃ 15M ⊃ 5M).
   910→           Раньше расширяли parent — это приводило к слиянию всех зон в одну
   911→           (матрёшка раскручивалась вверх: H1→H4→D1).
   912→        2. D1 зона ограничивается ±10% от текущей цены (LLM иногда берёт
   913→           исторический максимум за 100 свечей).
   914→        Порядок вложенности: 1D ⊃ 4H ⊃ 1H ⊃ 15M ⊃ 5M
   915→        """
   916→        if not tf_zones or price is None:
   917→            return tf_zones
   918→
   919→        nesting_order = ["1D", "4H", "1H", "15M", "5M"]
   920→        cap_pct = 0.10  # ±10% от цены
   921→
   922→        # 2a-1: D1 cap УБРАН. Был нужен для сырых экстремумов (57758-67255).
   923→        # Теперь LLM анализирует зоны по графикам + fallback VP/ZigZag —
   924→        # cap режет реалистичные зоны в сильных трендах (support на -30% обрезался до -10%).
   925→        # Оставлена только валидация lower>price (2a-1b ниже).
   926→
   927→        # 2a-1b: валидация для ВСЕХ ТФ — lower не выше цены, upper не ниже цены.
   928→        # Если lower > price → вся зона выше цены (XAUT D1 lower=4367 > price=4058).
   929→        #   Сдвигаем lower к цене: если upper тоже > price → зона [price*0.97, upper],
   930→        #   если upper был сдвинут D1 cap и стал близко к price → [price*0.97, price*1.03].
   931→        # Если upper < price → аналогично для upper.
   932→        margin_pct = 0.03  # ±3% от цены для минимальной зоны
   933→        for tf_key, z in tf_zones.items():
   934→            if not isinstance(z, dict):
   935→                continue
   936→            z_lower = z.get("lower")
   937→            z_upper = z.get("upper")
   938→            if z_lower is not None and z_lower > price:
   939→                z["lower"] = round(price * (1 - margin_pct), 2)
   940→            if z_upper is not None and z_upper < price:
   941→                z["upper"] = round(price * (1 + margin_pct), 2)
   942→            # После сдвигов: lower >= upper → вырожденная зона
   943→            if z.get("lower") is not None and z.get("upper") is not None and z["lower"] >= z["upper"]:
   944→                z["upper"] = round(price * (1 + margin_pct), 2)
   945→                z["lower"] = round(price * (1 - margin_pct), 2)
   946→
   947→        # 2a-2: валидация вложенности (от старшего к младшему)
   948→        for i in range(len(nesting_order) - 1):
   949→            parent_tf = nesting_order[i]
   950→            child_tf = nesting_order[i + 1]
   951→            parent = tf_zones.get(parent_tf)
   952→            child = tf_zones.get(child_tf)
   953→            if not isinstance(parent, dict) or not isinstance(child, dict):
   954→                continue
   955→
   956→            p_upper = parent.get("upper")
   957→            p_lower = parent.get("lower")
   958→            c_upper = child.get("upper")
   959→            c_lower = child.get("lower")
   960→
   961→            if p_lower is None and p_upper is None:
   962→                continue
   963→
   964→            # Если child выходит ЗА parent — СУЗИТЬ child до parent.
   965→            # Старший ТФ авторитетнее: D1 ⊃ 4H ⊃ 1H ⊃ 15M ⊃ 5M.
   966→            # Раньше расширяли parent → все зоны сливались в одну максимальную.
   967→            if c_lower is not None and p_lower is not None and c_lower < p_lower:
   968→                child["lower"] = p_lower
   969→            if c_upper is not None and p_upper is not None and c_upper > p_upper:
   970→                child["upper"] = p_upper
   971→
   972→        return tf_zones
   973→

   978→    # Если соседние ТФ имеют идентичные зоны (sticking):

   980→    #   - child структурная зона → confluence, НЕ трогать (общий уровень = сильный)
   981→    # Убрано: синтетическое расширение parent (expand_pct).
   982→    # -----------------------------
   983→    def _enforce_zone_uniqueness(tf_zones: dict, price: float | None) -> dict:
   984→        """
   985→        Sticking detection с двумя путями:
   986→        1) Child зона = микроканал → удалить (fallback подставит реальную).
   987→        2) Child зона структурная (>= min_span) → confluence (не трогать).
   988→
   989→        После top-down (T1-T5) общие границы = confluence = нормально.
   990→        Синтетическое расширение parent убрано.
   991→        """
   992→        if not tf_zones or price is None:
   993→            return tf_zones
   994→
   995→        tf_pairs = [("1D", "4H"), ("4H", "1H"), ("1H", "15M"), ("15M", "5M")]
   996→        tolerance_pct = 0.005  # 0.5% — зона считается "прилипшей"
   997→        min_span_pct = {"1D": 0.025, "4H": 0.020, "1H": 0.012, "15M": 0.008, "5M": 0.004}
   998→        price_safe = price if price > 0 else 1.0
   999→
  1000→        to_delete: list[str] = []
  1001→
  1002→        for parent_tf, child_tf in tf_pairs:
  1003→            parent = tf_zones.get(parent_tf)
  1004→            child = tf_zones.get(child_tf)
  1005→            if not isinstance(parent, dict) or not isinstance(child, dict):
  1006→                continue
  1007→            p_lower = parent.get("lower")
  1008→            p_upper = parent.get("upper")
  1009→            c_lower = child.get("lower")
  1010→            c_upper = child.get("upper")
  1011→            if p_lower is None or p_upper is None or c_lower is None or c_upper is None:
  1012→                continue
  1013→
  1014→            lower_diff = abs(p_lower - c_lower) / price_safe
  1015→            upper_diff = abs(p_upper - c_upper) / price_safe
  1016→            if lower_diff < tolerance_pct and upper_diff < tolerance_pct:
  1017→                child_span = abs(c_upper - c_lower) / price_safe
  1018→                child_min = min_span_pct.get(child_tf, 0.002)
  1019→                if child_span < child_min:
  1020→                    # Child зона = микроканал, не структурный range → удалить
  1021→                    logging.info(
  1022→                        "CONFLUENCE: %s sticking to %s but microchannel (%.3f%% < %.3f%%) — removing",
  1023→                        child_tf, parent_tf, child_span * 100, child_min * 100,
  1024→                    )
  1025→                    to_delete.append(child_tf)
  1026→                else:
  1027→                    # Структурная зона совпадает с parent → confluence, НЕ трогаем
  1028→                    logging.info(
  1029→                        "CONFLUENCE: %s matches %s (span=%.2f%%) — leaving as is",
  1030→                        child_tf, parent_tf, child_span * 100,
  1031→                    )
  1032→
  1033→        for tf_del in to_delete:
  1034→            del tf_zones[tf_del]
  1035→
  1036→        return tf_zones
  1037→
  1038→    data["tf_zones"] = _enforce_zone_uniqueness(data["tf_zones"], data.get("price"))

  1065→                del tf_zones[tf_key]
