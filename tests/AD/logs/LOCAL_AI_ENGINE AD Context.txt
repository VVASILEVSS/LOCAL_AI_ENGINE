//@version=5
indicator("LOCAL_AI_ENGINE A/D Context", overlay=false, max_labels_count=500, max_lines_count=500)

// ─────────────────────────────────────────────────────────────
// Назначение: внутренний A/D-контекст для LOCAL_AI_ENGINE.
// Отвечает за: линию A/D, CMF, дивергенции, подтверждение импульса и боковика.
// Связано с: Wyckoff, ZigZag, false breakout / retest фильтрацией.
// ─────────────────────────────────────────────────────────────

// ===== Inputs =====
groupMain = "Main"
cmfLen      = input.int(20,  "CMF Length", group=groupMain, minval=1)
emaLen      = input.int(21,  "A/D Smoothing EMA", group=groupMain, minval=1)
pivotLeft   = input.int(5,   "Pivot Left", group=groupMain, minval=1)
pivotRight  = input.int(5,   "Pivot Right", group=groupMain, minval=1)

groupLogic = "Logic"
showSignals     = input.bool(true,  "Show Divergence Labels", group=groupLogic)
showBgBias      = input.bool(true,  "Show Background Bias", group=groupLogic)
showVolumeState = input.bool(true,  "Show Volume State", group=groupLogic)

// ===== Core A/D =====
// Money Flow Multiplier * Volume
barRange = math.max(high - low, syminfo.mintick)
mfm      = ((close - low) - (high - close)) / barRange
mfv      = mfm * volume

ad_raw = ta.cum(mfv)
ad_ema = ta.ema(ad_raw, emaLen)

// ===== CMF =====
cmf_num = ta.cum(mfv) - ta.cum(mfv)[cmfLen]
cmf_den = ta.cum(volume) - ta.cum(volume)[cmfLen]
cmf_val = cmf_den != 0 ? cmf_num / cmf_den : 0.0

// ===== Helpers =====
isRising(x, len) =>
    x > x[len]

isFalling(x, len) =>
    x < x[len]

bias =
     ad_ema > ad_ema[1] and cmf_val > 0 ? 1 :
     ad_ema < ad_ema[1] and cmf_val < 0 ? -1 : 0

biasText =
     bias == 1 ? "Bullish volume bias" :
     bias == -1 ? "Bearish volume bias" : "Neutral / range"

// ===== Divergence logic =====
pl = ta.pivotlow(low, pivotLeft, pivotRight)
ph = ta.pivothigh(high, pivotLeft, pivotRight)

var float prevPriceLow = na
var float prevAdLow    = na
var float prevPriceHigh = na
var float prevAdHigh    = na

bullDiv = false
bearDiv = false

if not na(pl)
    currPriceLow = low[pivotRight]
    currAdLow    = ad_raw[pivotRight]
    if not na(prevPriceLow) and not na(prevAdLow)
        bullDiv := currPriceLow < prevPriceLow and currAdLow > prevAdLow
    prevPriceLow := currPriceLow
    prevAdLow    := currAdLow

if not na(ph)
    currPriceHigh = high[pivotRight]
    currAdHigh    = ad_raw[pivotRight]
    if not na(prevPriceHigh) and not na(prevAdHigh)
        bearDiv := currPriceHigh > prevPriceHigh and currAdHigh < prevAdHigh
    prevPriceHigh := currPriceHigh
    prevAdHigh    := currAdHigh

// ===== Volume state =====
volState =
     bias == 1 and cmf_val > 0.05 ? "Confirmation bullish" :
     bias == -1 and cmf_val < -0.05 ? "Confirmation bearish" :
     math.abs(cmf_val) < 0.02 ? "Weak / neutral" : "Mixed"

// ===== Plotting =====
hline(0, "Zero", color=color.new(color.gray, 80))

plot(ad_raw, "A/D Raw", color=color.new(color.teal, 0), linewidth=1)
plot(ad_ema, "A/D EMA", color=color.new(color.orange, 0), linewidth=2)

plot(cmf_val, "CMF", color=color.new(color.blue, 0), linewidth=2, style=plot.style_line)

// Background bias
bgcolor(showBgBias ? (bias == 1 ? color.new(color.green, 90) : bias == -1 ? color.new(color.red, 90) : color.new(color.gray, 95)) : na)

// Signals
if showSignals and bullDiv
    label.new(bar_index - pivotRight, ad_raw[pivotRight], "Bullish divergence", style=label.style_label_up, color=color.new(color.green, 0), textcolor=color.white)

if showSignals and bearDiv
    label.new(bar_index - pivotRight, ad_raw[pivotRight], "Bearish divergence", style=label.style_label_down, color=color.new(color.red, 0), textcolor=color.white)

// ===== Table =====
var table t = table.new(position.top_right, 1, 4, border_width=1)

if barstate.islast
    table.cell(t, 0, 0, "A/D bias: " + biasText, text_color=color.white, bgcolor=bias == 1 ? color.new(color.green, 40) : bias == -1 ? color.new(color.red, 40) : color.new(color.gray, 60))
    table.cell(t, 0, 1, "CMF(" + str.tostring(cmfLen) + "): " + str.tostring(cmf_val, format.mintick), text_color=color.white, bgcolor=color.new(color.blue, 70))
    table.cell(t, 0, 2, "Volume: " + volState, text_color=color.white, bgcolor=color.new(color.purple, 70))
    table.cell(t, 0, 3, "Div: " + (bullDiv ? "Bullish" : bearDiv ? "Bearish" : "None"), text_color=color.white, bgcolor=color.new(color.gray, 70))
